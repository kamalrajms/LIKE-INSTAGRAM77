<!-- // ================== Mobile Menu Toggle ================== -->
<script>
    const menuToggle = document.querySelector(".header-menu-toggle");
    const menuIcon = document.querySelector(".header-menu-icon");
    const closeIcon = document.querySelector(".header-close-icon");
    const sidebar = document.querySelector("aside");

    menuToggle.addEventListener("click", () => {
        sidebar.classList.toggle("active");

        if (sidebar.classList.contains("active")) {
            menuIcon.style.display = "none";
            closeIcon.style.display = "block";
        } else {
            menuIcon.style.display = "block";
            closeIcon.style.display = "none";
        }
    });
</script>

<!-- // ================== Mobile Dropdown Menu ================== -->
<script>
    // Target the HOME dropdown in mobile menu
    const mobileHomeDown = document.querySelector(".header-mobile-aside .home-down");
    
    if (mobileHomeDown) {
        mobileHomeDown.addEventListener("click", function(e) {
            e.preventDefault();
            
            // Find the parent li and then the dropdown ul
            const parentLi = this.closest(".home-menu");
            const dropdownUl = parentLi.querySelector(".header-mobile-option-ul");
            
            // Toggle display
            if (dropdownUl.style.display === "flex" || dropdownUl.style.display === "block") {
                dropdownUl.style.display = "none";
            } else {
                dropdownUl.style.display = "block";
            }
            
            // Optional: Rotate the dropdown icon
            const icon = this.querySelector("svg");
            if (icon) {
                icon.style.transform = dropdownUl.style.display === "block" ? "rotate(180deg)" : "rotate(0deg)";
                icon.style.transition = "transform 0.3s ease";
            }
        });
    }
    
    // Alternative: If you want to handle multiple dropdowns (if you add more in future)
    const mobileDropDowns = document.querySelectorAll(".header-mobile-aside .home-down");
    mobileDropDowns.forEach(dropDown => {
        dropDown.addEventListener("click", function(e) {
            e.preventDefault();
            
            const parentLi = this.closest(".home-menu");
            const option = parentLi.querySelector("ul");
            
            if (option.style.display === "flex" || option.style.display === "block") {
                option.style.display = "none";
            } else {
                option.style.display = "block";
            }
            
            // Toggle icon rotation
            const icon = this.querySelector("svg");
            if (icon) {
                icon.style.transform = option.style.display === "block" ? "rotate(180deg)" : "rotate(0deg)";
                icon.style.transition = "transform 0.3s ease";
            }
        });
    });
    
    // Close mobile menu when clicking a link (optional)
    const mobileNavLinks = document.querySelectorAll(".header-mobile-nav-link:not(.home-down)");
    mobileNavLinks.forEach(link => {
        link.addEventListener("click", function() {
            sidebar.classList.remove("active");
            menuIcon.style.display = "block";
            closeIcon.style.display = "none";
        });
    });
</script>

<!-- // ================== Scroll Animations, Tabs, Slider, FAQ ================== -->
<script>
    // ===== SCROLL ANIMATIONS =====
    var revealElements = document.querySelectorAll('.reveal');

    var revealObserver = new IntersectionObserver(function(entries) {
        entries.forEach(function(entry) {
            if (entry.isIntersecting) {
                entry.target.classList.add('visible');
            }
        });
    }, { threshold: 0.1, rootMargin: '0px 0px -40px 0px' });

    revealElements.forEach(function(el) {
        revealObserver.observe(el);
    });

    // ===== PROGRESS BAR ANIMATION =====
    var progressBar = document.getElementById('progressBar');
    if (progressBar) {
        setTimeout(function() {
            progressBar.style.width = '78%';
        }, 500);
    }

    // ===== TAB NAVIGATION =====
    var tabLinks = document.querySelectorAll('.tab-nav a');
    var tabContents = document.querySelectorAll('.tab-content');

    tabLinks.forEach(function(link) {
        link.addEventListener('click', function(e) {
            e.preventDefault();
            var tabId = this.getAttribute('data-tab');

            // Remove active from all tabs
            tabLinks.forEach(function(t) { t.classList.remove('active'); });
            tabContents.forEach(function(c) { c.classList.remove('active'); });

            // Add active to clicked tab
            this.classList.add('active');
            document.getElementById('tab-' + tabId).classList.add('active');

            // Re-trigger reveal animations for new tab content
            var newReveals = document.getElementById('tab-' + tabId).querySelectorAll('.reveal');
            newReveals.forEach(function(el) {
                el.classList.remove('visible');
                setTimeout(function() {
                    el.classList.add('visible');
                }, 100);
            });
        });
    });

    // ===== TESTIMONIAL SLIDER =====
    var slider = document.getElementById('slider');
    var prevBtn = document.getElementById('prevBtn');
    var nextBtn = document.getElementById('nextBtn');
    var scrollAmount = 375;

    if (prevBtn && nextBtn && slider) {
        prevBtn.addEventListener('click', function(e) {
            e.preventDefault();
            slider.scrollBy({ left: -scrollAmount, behavior: 'smooth' });
        });

        nextBtn.addEventListener('click', function(e) {
            e.preventDefault();
            slider.scrollBy({ left: scrollAmount, behavior: 'smooth' });
        });
    }

    // ===== FAQ ACCORDION =====
    var faqItems = document.querySelectorAll('.faq-item');

    faqItems.forEach(function(item) {
        var question = item.querySelector('.faq-question');
        if (question) {
            question.addEventListener('click', function() {
                var isActive = item.classList.contains('active');

                // Close all
                faqItems.forEach(function(faq) {
                    faq.classList.remove('active');
                });

                // Open clicked (if it wasn't active)
                if (!isActive) {
                    item.classList.add('active');
                }
            });
        }
    });
</script>
